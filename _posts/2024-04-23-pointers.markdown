# Указатели

```c
#include <stdio.h>
 
int main(void) 
{
         char d = 10;
         char *gpt;
         gpt = &d;
 
         printf("gpt = %p, *gpt = %d, d = %d\n", gpt, *gpt, d);
         
         *gpt = 100;
         printf("gpt = %p, *gpt = %d, d = %d\n", gpt, *gpt, d);
 
         return 0;
}
```

```c
#include <stdio.h>
 
int main(void) 
{
         int arg = 7;
         int *ptr_arg, *ptr;
 
         ptr_arg = &arg;
         ptr = ptr_arg;
 
         printf("*ptr = %d, arg = %d\n", *ptr, arg);
 
         *ptr_arg = 100;
         printf("*ptr = %d, arg = %d\n", *ptr, arg);
 
         return 0;
}
```
