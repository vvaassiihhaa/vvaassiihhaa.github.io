# while

```c
#include <stdio.h>

int main( void ) {
  int s = 0, x = 1;

  while( scanf( "%d", &x ) == 1 && x != 0 )
    s += x;

  printf( "s = %d\n", s );

  return 0;
}
```

# Условная компиляция

```c
#define LANG_C
 
#if defined(LANG_C)
#   include <stdio.h>
#else
#   include <iostream>
#endif
 
int main( void )
{
         int x=5;
#ifdef LANG_C
         printf("%d\n", x);
#else
         std::cout << x << std::endl;
#endif
 
    return 0;
}
```
